\documentclass{standalone}

\usepackage{msc}
\begin{document}

\begin{msc}[small values, instance
distance=7.6cm]{A protocol that makes use of certificates}

\declinst{prof}{ }{Professor}
\declinst{student}{}{Paul}

\mess{I'm your professor, use this certificate: $Cert_{Prof}$ 
}{prof}{student}
\nextlevel
\action*{Check signature of $Cert_{Prof}$}{student}
\nextlevel
\nextlevel
\nextlevel
\action*{Extract $pk_{Prof}$ from $Cert_{Prof}$}{student}
\nextlevel
\nextlevel
\nextlevel
\mess{$Cert_{Paul}$, $\{$$\{$Hi professor, this is my secret...$\}_{pk_{Prof}}$ 
$\}_{sk_{Paul}}$}{student}{prof}
\nextlevel
\action*{Check signature of $Cert_{Paul}$}{prof}
\nextlevel
\nextlevel
\nextlevel
\action*{Extract $pk_{Paul}$ from $Cert_{Paul}$}{prof}
\nextlevel
\nextlevel
\nextlevel
\action*{Decrypt message, first with $pk_{Paul}$ then $sk_{Prof}$}{prof}
\nextlevel
\nextlevel
\nextlevel
\end{msc}

\end{document}
